مــــــا هـــــــي خـــــدمـــــــــة مجتمع نقــــاء البيـئيــــــــــــة لاستبــــــــــدال زيـــــت القلــي المستعمـــــــل؟

تقوم خدمة نقاء البيئية لإعادة تدوير زيوت الطعام المستعمل على شراء هذه الزيوت من المنازل وذلك يكون بهدية مادية للعميل مقابل الكيلو الواحد من الزيت وهذا ليكون حافز للعملاء للإشتراك في الخدمة والتخلص الآمن من هذه الزيوت بدلا من التخلص منها في أحواض المطابخ أو القمامة مما يؤدي إلى أضرار بالبيئة والصرف الصحي.

جـــــــائـــــــزة ابتكـــــــــــــــار خانــــــــــــــــة لريــــــــــــــادة الأعمــــــــــــال

فـاز مجتمع نقـاء بالمركز الأول بجائـــزة 55 ألـــف جنيـــه في مســابقة إبتكـار خانــة لريادة الأعمــل عـــام 2020 لأفضـــل مشروع ذا مساهمة اجتمــاعية يعمـل علـى حـل مشكلة جذرية في المجتمع.
